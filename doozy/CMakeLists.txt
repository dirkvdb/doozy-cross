CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
CMAKE_POLICY(VERSION 2.8)
INCLUDE(ExternalProject)

SET(NAME doozy)

EXTERNALPROJECT_ADD(${NAME}
    GIT_REPOSITORY https://github.com/dirkvdb/doozy.git
    GIT_TAG develop
    GIT_SUBMODULES
    INSTALL_DIR ${CMAKE_BINARY_DIR}/local
    CMAKE_COMMAND cmake -DSTATIC_BINARY=ON -DTESTTOOLS=ON -DCMAKE_BUILD_TYPE=Release -DCMAKE_INSTALL_PREFIX=<INSTALL_DIR> -DCMAKE_TOOLCHAIN_FILE=${CMAKE_TOOLCHAIN_FILE}
)
